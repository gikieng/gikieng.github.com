10 Point p[100005];
11 int cmp(const void *a,const void *b)
12 {
13 if((*(Point *)a).y==(*(Point *)b).y)
14     return (*(Point *)b).x>(*(Point *)a).x?-1:1;
15     return (*(Point *)b).y>(*(Point *)a).y?-1:1;
16 }
17 int cmp1(const void *a,const void *b)
18 {
19 if((*(Point *)a).x==(*(Point *)b).x)
20     return (*(Point *)b).y>(*(Point *)a).y?-1:1;
21     return (*(Point *)b).x>(*(Point *)a).x?-1:1;
22 }
