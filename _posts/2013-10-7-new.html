 1 //注意输入1 0 和0 1时是没有输出的,还要控制换行和空格；
 2 #include<stdio.h>
 3 int a[100];
 4 int n,l;
 5 int flag,cnt;
 6 void dfs(int cur)
 7 {
 8     int i,j,k;
 9     if(cnt++==n)
10     {
11         for(i=0;i<cur;i++)
12         {
13             printf("%c",a[i]+'A');
14             if(i!=cur-1)
15             {
16                 if((i+1)%64==0)
17                     printf("\n");
18                 else if((i+1)%4==0)
19                     printf(" ");
20             }
21         }
22         if(cur!=0)
23         printf("\n%d\n",cur);
24         flag=0;
25         return;
26     }
27     for(i=0;i<l;i++)
28     {
29         a[cur]=i;
30         int ok=0;
31         for(j=1;2*j<=cur+1;j++)
32         {
33             ok=1;
34             for(k=0;k<j;k++)
35             {
36                 if(a[cur-k]!=a[cur-k-j])
37                     {
38                         ok=0;
39                         break;
40                     }
41             }
42             if(ok==1)
43                 break;
44         }
45             if(ok==0&&flag)
46                 dfs(cur+1);
47     }
48 }
49 int main()
50 {
51     while(~scanf("%d%d",&n,&l))
52     {
53         cnt=0;
54         flag=1;
55         dfs(0);
56     }
57 }
